      match /addresses/{addressId} {

    }

    // ğŸ›ï¸ Products â€“ Ã¶ffentlich lesbar, Admin darf alles

        request.resource.data.diff(resource.data).affectedKeys().hasOnly(['stock', 'status']);
    }
  }
